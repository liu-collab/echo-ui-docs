# echo-ui

基于 Vue3 的组件库，样式借用 element-plus 的，学习使用

# 安装

npm install -S echo-echonpm-ui

# 引入

import EchoUI from 'echo-echonpm-ui';
import 'echo-echonpm-ui/lib/echo-echonpm-ui.css';

createApp(App).use(EchoUI);
